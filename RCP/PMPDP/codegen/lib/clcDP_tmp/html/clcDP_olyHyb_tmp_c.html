  440             /*  kinetsiche Energie des betrachten(consider) VorgÃ¤ngerspunkts */
  441             /*  bestimmen(determine) */
  442             /*    determine the kinetic energy of the previous path_idx */
  443             /*  PrÃ¼fen(check), ob eine erlaubte Beschleunigung vorliegt. */
  444             /*  Ansonsten zum nÃ¤chsten Schleifendurchlauf springen */
  445             /*    Check whether an allowable acceleration exists. */
  446             /*    Otherwise, jump to the next iteration */
  447             vehAcc = (engKinAct - engKinMat_engKinInx_wayInx[engKinPreInx + 11 *
  448                       ((int)b_wayInx - 2)]) / vehMas / wayStp;
  449             if ((vehAcc < vehAccMin) || (vehAcc > vehAccMax)) {
  450             } else {
  451               /*  Schleife Ã¼ber allen ZustÃ¤nde (relativer Index) */
  452               /*    Loop through all the states (relative index) */
  453               i1 = (int)(staPreMax + (1.0 - (double)staPreMin));
  454               for (staPre = 0; staPre < i1; staPre++) {
  455                 b_staPre = (double)staPreMin + (double)staPre;
  456   
  457                 /*  CURRENT GEAR CHANGE LOOP */
  458                 /*                     %% Batterieenergie beim betrachteten VorgÃ¤nger */
  459                 /*  battery energy when considering last path_idx */
  460                 /*    note: batengPreMat has dims #_KE_states x #_gears */
  461                 /*  Sollte es keinen gÃ¼ltigen VorgÃ¤nger geben, wird zum */
  462                 /*  nÃ¤chsten Schleifendurchlauf gesprungen */
  463                 /*    if there is no valid previous battery energy, jump */
  464                 /*    to the next loop iteration */
  465                 if (rtIsInf(batEngPreMat->data[engKinPreInx + batEngPreMat->size[0]
  466                             * ((int)b_staPre - 1)])) {
  467                 } else {
  468                   /*                     %% Antriebsstrangzustand und Strafkosten bestimmen    */
  469                   /*    determine gear and penalty costs */
  470                   /*  Kosten fÃ¼r Zustandswechsel setzen */
  471                   /*    set costs for state changes */
  472                   if (1.0 + (double)b_loop_ub == b_staPre) {
  473                     /*  Entspricht der VorgÃ¤ngerzustand dem aktuellen  */
  474                     /*  Zustand werden keine Kosten gesetzt */
  475                     /*    staying in current state? set penalty cost to 0 */
  476                     staChgPenCos = 0.0;
  477                   } else {
  478                     /*  Ansonsten einfache Zustandswechselkosten */
  479                     /*  berechnen */
  480                     /*    otherwise apply a penalty cost to changing gear */
  481                     staChgPenCos = staChgPenCosVal;
  482   
  483                     /* <-penCos is input */
  484                   }
  485   
  486                   /*                     %% Berechnung der optimalen Kosten zum aktuellen Punkt */
  487                   /*    calculating optimal cost to the current point */
  488                   /*  externe Funktion ausfÃ¼hren, die minimale Kosten der */
  489                   /*  Hamiltonfunktion zurÃ¼ckgibt */
  490                   /*    run the min-cost Hamiltonian finding function  */
  491                   clcPMP_olyHyb_tmp(engKinMat_engKinInx_wayInx[engKinPreInx + 11 *
  492                                     ((int)b_wayInx - 2)], engKinAct, 1.0 + (double)
  493                                     b_loop_ub, slpVec_wayInx[(int)b_wayInx - 2],
  494                                     batEngPreMat->data[engKinPreInx +
  495                                     batEngPreMat->size[0] * ((int)b_staPre - 1)],
  496                                     psiBatEng, psiTim, batPwrAux, batEngStp,
  497                                     wayStp, FZG, &cosHam, &batFrc, &fulFrc);
  498   
  499                   /*                      % minimale Kosten der Hamiltonfunktion zum aktuellen */
  500                   /*                      % Punkt bestimmen */
  501                   /*                      [cosHamMin,optPreInx] ... */
  502                   /*                          = min([cosHam... */
  503                   /*                          + cos2goPreMat(engKinPreInx,staPre)... */
  504                   /*                          + staChgPenCos... */
  505                   /*                          ,cosHamMin]); %#ok<PFBNS> */
  506                   /*  combine the min hamil. cost w/ previous costs and  */
  507                   /*    gear penalty to get current cost */
  508                   cosAct = (cosHam + cos2goPreMat->data[engKinPreInx +
  509                             cos2goPreMat->size[0] * ((int)b_staPre - 1)]) +
  510                     staChgPenCos / wayStp;
  511   
  512                   /*  Wenn der aktuelle Punkt besser ist, als der in */
  513                   /*  cosHamMin gespeicherte Wert, werden die AusgabegrÃ¶ÃŸen */
  514                   /*  neu beschrieben. */
  515                   /*    if current point is better than the cost value */
  516                   /*    stored in CosHamMin, then rewrite the output */
  517                   if (cosAct < cosHamMin) {
  518                     cosHamMin = cosAct;
  519   
  520                     /*  new hamil. cost */
  521                     staPreOptInx = (int)b_staPre;
  522   
  523                     /*  new optimal gear idx */
  524                     engKinPreOptInx = 1.0 + (double)engKinPreInx;
  525   
  526                     /*  new optimal KEidx */
  527                     batFrcOpt = batFrc;
  528   
  529                     /*  new optimal battery force */
  530                     /*  new opt. battery energy = (batt. force * */
  531                     /*  displacement diff) + previous battery energy valu */
  532                     batEngOpt = batFrc * wayStp + batEngPreMat->data[engKinPreInx
  533                       + batEngPreMat->size[0] * ((int)b_staPre - 1)];
  534   
  535                     /*  new opt. fuel energy = (fuel force * displacement */
  536                     /*  diff) + previous fuel energy value */
  537                     fulEngOpt = fulFrc * wayStp + fulEngPreMat->data[engKinPreInx
  538                       + fulEngPreMat->size[0] * ((int)b_staPre - 1)];
  539                   }
  540                 }
  541               }
  542   
  543               /*  end of gear changes loop */
  544             }
  545           }
  546   
  547           /*  end of running through previous KE states loop */
  548           if (!rtIsInf(cosHamMin)) {
  549             /*  optimale Kosten zum aktuellen Punkt speichern */
  550             /*    save min hamilton value for current point */
  551             cos2goActMat->data[loop_ub + cos2goActMat->size[0] * b_loop_ub] =
  552               cosHamMin;
  553   
  554             /*  optimale Batterieenergie zum aktuellen Punkt speichern */
  555             /*    save optimal battery energy for current point */
  556             batEngActMat->data[loop_ub + batEngActMat->size[0] * b_loop_ub] =
  557               batEngOpt;
  558   
  559             /*  optimale Krafstoffenergie zum aktuellen Punkt speichern */
  560             /*    save optimal fuel energy for current point */
  561             fulEngActMat->data[loop_ub + fulEngActMat->size[0] * b_loop_ub] =
  562               fulEngOpt;
  563   
  564             /*  optimale Batterieenergie zum aktuellen Punkt */
  565             /*  FlussgrÃ¶ÃŸe gilt im Intervall */
  566             /*    populate optimal battery energy flux quantity at point  */
  567             /*    that's applicable to current interval */
  568             batFrcOptMat->data[loop_ub + batFrcOptMat->size[0] * b_loop_ub] =
  569               batFrcOpt;
  570   
  571             /*  optimalen VorgÃ¤nger codieren Ã¼ber Funktion sub2ind */
  572             /*  und speichern im Tensor */
  573             /*    opt. predecessor idx encoding w/ sub2ind, store in Tn3 */
  574             optPreInxTn3->data[(loop_ub + optPreInxTn3->size[0] * b_loop_ub) +
  575               optPreInxTn3->size[0] * optPreInxTn3->size[1] * ((int)b_wayInx - 1)]
  576               = (int)engKinPreOptInx + (int)engKinNum * (staPreOptInx - 1);
  577           }
  578   
  579           /*  end of ~inf(hamiltonian) if-statement */
  580         }
  581   
  582         /*  end of looping through all gears */
  583       }
  584   
  585       /*  end of looping through all the current kinetic energy states */
  586       /*  Speichern der Batterieenergie fÃ¼r den nÃ¤chsten Schleifendurchlauf */
  587       /*    save battery energy value as previous path_idx val for next loop  */
  588       i1 = batEngPreMat->size[0] * batEngPreMat->size[1];
  589       batEngPreMat->size[0] = batEngActMat->size[0];
  590       batEngPreMat->size[1] = batEngActMat->size[1];
  591       emxEnsureCapacity((emxArray__common *)batEngPreMat, i1, (int)sizeof(double));
  592       loop_ub = batEngActMat->size[0] * batEngActMat->size[1];
  593       for (i1 = 0; i1 < loop_ub; i1++) {
  594         batEngPreMat->data[i1] = batEngActMat->data[i1];
  595       }
  596   
  597       /*  Speichern der Krafstoffenergie fÃ¼r den nÃ¤chsten Schleifendurchlauf */
  598       /*    save fuel energy value as previous path_idx value for the next loop */
  599       i1 = fulEngPreMat->size[0] * fulEngPreMat->size[1];
  600       fulEngPreMat->size[0] = fulEngActMat->size[0];
  601       fulEngPreMat->size[1] = fulEngActMat->size[1];
  602       emxEnsureCapacity((emxArray__common *)fulEngPreMat, i1, (int)sizeof(double));
  603       loop_ub = fulEngActMat->size[0] * fulEngActMat->size[1];
  604       for (i1 = 0; i1 < loop_ub; i1++) {
  605         fulEngPreMat->data[i1] = fulEngActMat->data[i1];
  606       }
  607   
  608       /*  Speichern der Kosten fÃ¼r den nÃ¤chsten Schleifendurchlauf */
  609       /*    save cost as previous path_idx value for the next loop */
  610       i1 = cos2goPreMat->size[0] * cos2goPreMat->size[1];
  611       cos2goPreMat->size[0] = cos2goActMat->size[0];
  612       cos2goPreMat->size[1] = cos2goActMat->size[1];
  613       emxEnsureCapacity((emxArray__common *)cos2goPreMat, i1, (int)sizeof(double));
  614       loop_ub = cos2goActMat->size[0] * cos2goActMat->size[1];
  615       for (i1 = 0; i1 < loop_ub; i1++) {
  616         cos2goPreMat->data[i1] = cos2goActMat->data[i1];
  617       }
  618   
  619       /*  optimale Kraftstoffenergie zum aktuellen Punkt */
  620       /*    optimal fuel energy at current point - save current mat in tensor */
  621       loop_ub = fulEngActMat->size[1];
  622       for (i1 = 0; i1 < loop_ub; i1++) {
  623         b_loop_ub = fulEngActMat->size[0];
  624         for (engKinPreInx = 0; engKinPreInx < b_loop_ub; engKinPreInx++) {
  625           fulEngOptTn3->data[(engKinPreInx + fulEngOptTn3->size[0] * i1) +
  626             fulEngOptTn3->size[0] * fulEngOptTn3->size[1] * ((int)b_wayInx - 1)] =
  627             fulEngActMat->data[engKinPreInx + fulEngActMat->size[0] * i1];
  628         }
  629       }
  630   
  631       /*  optimale Batterieenergie zum aktuellen Punkt */
  632       /*    optimal battery force at current point - save current mat in tensor */
  633       /*  FlussgrÃ¶ÃŸe gilt im Intervall */
  634       /*    flux quantity applied over the interval */
  635       loop_ub = batFrcOptMat->size[1];
  636       for (i1 = 0; i1 < loop_ub; i1++) {
  637         b_loop_ub = batFrcOptMat->size[0];
  638         for (engKinPreInx = 0; engKinPreInx < b_loop_ub; engKinPreInx++) {
  639           batFrcOptTn3->data[(engKinPreInx + batFrcOptTn3->size[0] * i1) +
  640             batFrcOptTn3->size[0] * batFrcOptTn3->size[1] * ((int)b_wayInx - 2)] =
  641             batFrcOptMat->data[engKinPreInx + batFrcOptMat->size[0] * i1];
  642         }
  643       }
  644   
  645       /*  Ausgabe des aktuellen Schleifendurchlaufs */
  646       /*    output for current loop - print to terminal */
  647       if (disFlg != 0.0) {
  648         d_fprintf(b_wayInx - wayInxBeg, (b_wayInx - wayInxBeg) / (wayInxEnd -
  649                    wayInxBeg) * 100.0);
  650       }
  651   
  652       wayInx++;
  653     }
  654   
  655     emxFree_real_T(&fulEngActMat);
  656     emxFree_real_T(&batEngActMat);
  657     emxFree_real_T(&batFrcOptMat);
  658     emxFree_real_T(&fulEngPreMat);
  659     emxFree_real_T(&batEngPreMat);
  660     emxFree_real_T(&cos2goPreMat);
  661   
  662     /*  end of looping through all discretized path indexes */
  663     /*  END OF FUNCTION */
  664   }
