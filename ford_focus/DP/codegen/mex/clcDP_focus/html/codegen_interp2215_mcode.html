1%#codegen
 2% zi = codegen_interp2(X,Y,Z,xi,yi) gives the same result as
 3% interp2(X,Y,Z,xi,yi)
 4% Unlike interp2, codegen_interp2 is compatible with code generation
 5% Only linear interpolation is available
 6
 7% Usage restrictions
 8%   X and Y must have the same size as Z
 9%   e.g.,  [X,Y] = meshgrid(x,y);
10
11function Zi = codegen_interp2(X,Y,Z,xi,yi)
12
13ndx = 1/(X(1,2)-X(1,1));        ndy = 1/(Y(2,1)-Y(1,1));
14
15idyi=(xi - X(1,1))*ndx+1;       idxi=(yi - Y(1,1))*ndy+1;
16
17if idxi <= 1 || idyi <= 1 || idxi > length(Y(:,1)) || idyi > length(X(1,:)) ...
18    || isnan(xi) || isnan(yi)
19    Zi = inf;
20    return;
21end
22
23if idxi/ceil(idxi)~=1&&idyi/ceil(idyi)~=1
24    Z1=Z(ceil(idxi)-1,ceil(idyi)-1);
25    Z2=Z(ceil(idxi)-1,ceil(idyi));
26    Z3=Z(ceil(idxi),ceil(idyi)-1);
27    Z4=Z(ceil(idxi),ceil(idyi));
28    
29    Zi= Z1*(ceil(idxi)-idxi)*(ceil(idyi)-idyi)+...
30        Z2*(ceil(idxi)-idxi)*(1-(ceil(idyi)-idyi))+...
31        Z3*(1-(ceil(idxi)-idxi))*(ceil(idyi)-idyi)+...
32        Z4*(1-(ceil(idxi)-idxi))*(1-(ceil(idyi)-idyi));
33    
34elseif idxi/ceil(idxi)~=1&&idyi/ceil(idyi)==1
35    Z1=Z(ceil(idxi)-1,idyi);
36    Z2=Z(ceil(idxi),idyi);
37    
38    Zi=Z1*(ceil(idxi)-idxi)+Z2*(1-(ceil(idxi)-idxi));
39    
40elseif idxi/ceil(idxi)==1&&idyi/ceil(idyi)~=1    
41    Z1=Z(idxi,ceil(idyi)-1);
42    Z2=Z(idxi,ceil(idyi));
43    
44    Zi=Z1*(ceil(idyi)-idyi)+Z2*(1-(ceil(idyi)-idyi));
45else
46    Zi=Z(idxi,idyi);    
47end
48    
49end
