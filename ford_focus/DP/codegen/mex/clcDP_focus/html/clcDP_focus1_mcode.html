604                                + cos2goPreTn3(engStaPre+1, geaStaPre, batStaPreIdx_noEmo) ...

656                    optPreInxTn4(engStaAct+1,geaStaAct,batStaActInx,timInx) = ...
657                        sub2ind([engNum,geaNum, batNum],...
658                        engStaPreOptInx,geaStaPreOptInx, batStaPreOptInx);
659                end % end of ~inf(hamiltonian) if-statement
660            end %end of looping through all battery states
661            
662            fprintf('.');
663        end % end of looping through all gears

669%     batEngPreInxVec = batStaLimBot : batStaLimTop;
