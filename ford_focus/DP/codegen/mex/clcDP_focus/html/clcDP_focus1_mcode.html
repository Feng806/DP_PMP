493%                       % emoTrqMaxPosMat(:, geaStaPre);

508                            % zu hoch im hybridischen
509                            % Modus

515                            % Falls die Drehzahl des Verbrennungsmotors niedriger als die

519                            end

522                                % value of previous idx engine control state
523                                %  - unnecessary to assign - indexes represent bool values
524                                   % loop through all possible previous battery levels
525                                % calculate index for bat state - can move
526                                % outside for loop for improving computation tim later
527

529                                % the loop batStaPreIdx_counter, and because
530                                % we need batStaPreIdx to be used for
531                                % referencing vector indexes later on, the
532                                % variable batStaPreIdx will be used as an
533                                % index vector, and not as a scaled energy
534                                % level value index vector.
535                                % therefore, we subtract 1 from the index
536                                % when transforming batStaPreIdx into its
537                                % rescaled form batStaPre.
538                                batStaPreIdx = batStaPreIdxVec(batStaPreIdx_counter);
539                                batStaPre    = (batStaPreIdx-1) * batStaStp;
540                                
541                                % calculate E'
542                                batPwr = batStaAct - batStaPre;
543
544                                % calculate battery resistance
545                                % the code below is performing this code snippet across the vector
546                                % if batPwr < 0
547                                %     batRst = fzg_scalar_struct.batRstDch;
548                                % else
549                                %     batRst = fzg_scalar_struct.batRstChr;
550                                % end
551                                batRst = (batPwr < 0) * fzg_scalar_struct.batRstDch ...
552                                      + (batPwr >= 0) * fzg_scalar_struct.batRstChr;
553                                    
554                                % lookup previous bat state's voltage value
555                                batOcvPre = batOcv(batStaPreIdx);
556                                
557                                %% Berechnung der optimalen Kosten zum aktuellen Punkt
558                                %   calculating optimal cost to the current point
559
560                                % do it tim interval at a tim? will remove vector
561                                % aspects
562                                [minFul, emoTrq, iceTrq, brkTrq] =...
563                                    optTrqSplit_focus   ...
564                                    (                   ...
565                                    inputparams.brkBool,...
566                                    batPwr,             ...
567                                    batOcvPre,          ...
568                                    batRst,             ...
569                                    crsSpdPre,          ...
570                                    crsTrqPre,          ...
571                                    emoTrqMinPos,       ...
572                                    emoTrqMaxPos,       ...
573                                    emoPwrMinPos,       ...
574                                    emoPwrMaxPos,       ...
575                                    iceTrqMaxPos,       ...
576                                    iceTrqMinPos,       ...
577                                    crsSpdHybMax,       ... % maximum crankshaft rotational speed
578                                    crsSpdHybMin,       ... % minimum crankshaft rotational speed
579                                    inputparams.timStp, ...
580                                    inputparams.batPwrAux,...
581                                    vehVelVec,          ...
582                                    fzg_scalar_struct,  ...
583                                    fzg_array_struct    ...
584                                    );
585
586                                % combine the min hamil. cost w/ previous costs and 
587                                %   gear penalty to get current cost
588                                fulActTn3(engStaPre+1, geaStaPre,batStaPreIdx) ...
589                                    = minFul...
590                                    + cos2goPreTn3(engStaPre+1,geaStaPre, batStaPreIdx)...
591                                    + geaStaChgPenCos/inputparams.timStp  ...
592                                    + engStaChgPenCos/inputparams.timStp;
593                                
594                                emoTrqPreTn3(engStaPre+1, geaStaPre,batStaPreIdx) = emoTrq;
595                                iceTrqPreTn3(engStaPre+1, geaStaPre,batStaPreIdx) = iceTrq;
596                                brkTrqPreTn3(engStaPre+1, geaStaPre,batStaPreIdx) = brkTrq;
597
598                            end % end of bat energy changing loop
599                                
600                        else
601                            % ---- electric motor MUST satisfy crsTrq -----
602                            % IF ENGINE IS OFF
603                            % Prüfen, ob die Drehzahlgrenze des Elektromotors eingehalten wird
604                            %   check if electric motor speed limit is maintained
605                            if (crsSpdPreVec(geaStaPre) > crsSpdEmoMax); continue;
606                            end
607                            
608                            % determine previous state if engine waas off
609                            % based on batPwrDemIdxTn3 (a preprocessed tn3
610                            % containing amnt battery pwr needed to satisfy
611                            % a crankshaft power demand for all state
612                            % permutations)
613                            batStaPreIdx_noEmo = batStaActInx + ... - 1
614                                                 batPwrDemIdxTn3(timInx, batStaActInx, geaStaAct);
615
616                            % check your bounds 
617                            if (batStaPreIdx_noEmo < batStaMin/batStaStp + 1) || ...
618                                (batStaPreIdx_noEmo > batStaMax/batStaStp + 1)
619                                continue;
620                            end
621                            
622                            % penalty to get current cost
623                            fulActTn3(engStaPre+1, geaStaPre, batStaPreIdx_noEmo) ...
624                                = ... minFul ...
625                                + cos2goPreTn3(engStaPre+1, geaStaPre, batStaPreIdx_noEmo) ...
626                                + geaStaChgPenCos / inputparams.timStp ...
627                                + engStaChgPenCos / inputparams.timStp;
628
629                            emoTrqPreTn3(engStaPre+1, geaStaPre,batStaPreIdx_noEmo) = crsTrqPre;
630
631                            % brake torque in case of torque overshoot
632                            if inputparams.brkBool

636                                    brkTrqPreTn3(engStaPre+1, geaStaPre, batStaPreIdx_noEmo) = ...
637                                        (batPwrPre - crsPwrPre)/crsSpdPre;
638                                end
639                            end
640                            
641                        end % end of engStaPre condition check
642                        
643                    end % end of gear changes loop
644                end % end of running through previous engine state ctrl loop

646

648                [colmin, colminidx] = min(fulActTn3);
649                [matmin, matminidx] = min(colmin);
650                [minFulMin, batStaPreOptInx] = min(matmin);
651                geaStaPreOptInx     = matminidx(batStaPreOptInx);
652                engStaPreOptInx     = colminidx(:, geaStaPreOptInx, batStaPreOptInx);
653%                 engStaPreOptInx = batStaPreIdxVec(engStaPreInx);
654                
655                if ~isinf(minFulMin)

657                    %   tim diff) + previous battery energy valu
658                    %   - NOTE: batFrc*timStp calc is the same as the
659                    %       batFrc calculation in batFrcClc_a()
660                    %   -   why not output that calculation instead?
661                    batEngOpt = (batStaPreOptInx-1) * batStaStp; % + ...
662%                         batEngPreTn3(engStaPreOptInx,geaStaPreOptInx,batStaPreOptInx);
663                    
664                    % new opt. fuel energy = (fuel force * tim diff)
665                    %   + previous fuel energy value
666                    fulEngOpt = minFulMin + ...
667                    fulEngPreTn3(engStaPreOptInx,geaStaPreOptInx,batStaPreOptInx);%%#ok<PFBNS>
668                
669                    % optimale Kosten zum aktuellen Punkt speichern
670                    %   save min hamilton value for current point
671                    cos2goActTn3(engStaAct+1,geaStaAct,batStaActInx) = minFulMin;
672
673                    % optimale Batterieenergie zum aktuellen Punkt speichern
674                    %   save optimal battery energy for current point
675                    batEngActTn3(engStaAct+1,geaStaAct,batStaActInx) = batEngOpt;
676
677                    % optimale Krafstoffenergie zum aktuellen Punkt speichern
678                    %   save optimal fuel energy for current point
679                    fulEngActTn3(engStaAct+1,geaStaAct,batStaActInx) = fulEngOpt;
680
681                    % optimale Batterieenergie zum aktuellen Punkt
682                    % Flussgrï¿½ï¿½e gilt im Intervall
683                    %   populate optimal battery energy flux quantity at point 
684                    %   that's applicable to current interval
685                    batPwrOptTn3(engStaAct+1,geaStaAct,batStaActInx) = batStaActInx - batStaPreOptInx;
686                    
687                    % save optimal torque values for the given state
688                    % emoTrq
689                    emoTrqActTn3(engStaAct+1,geaStaAct,batStaActInx) = ...
690                        emoTrqPreTn3(engStaPreOptInx, geaStaPreOptInx, batStaPreOptInx);
691                    % iceTrq
692                    iceTrqActTn3(engStaAct+1,geaStaAct,batStaActInx) = ...
693                        iceTrqPreTn3(engStaPreOptInx, geaStaPreOptInx, batStaPreOptInx);
694                    % brkTrq
695                    brkTrqActTn3(engStaAct+1,geaStaAct,batStaActInx) = ...
696                        brkTrqPreTn3(engStaPreOptInx, geaStaPreOptInx, batStaPreOptInx);
697                    
698                    % optimalen Vorgänger codieren über Funktion sub2ind
699                    % und speichern im Tensor
700                    %   opt. predecessor idx encoding w/ sub2ind, store in Tn3
701                    optPreInxTn4(engStaAct+1,geaStaAct,batStaActInx,timInx) = ...
702                        sub2ind([engNum,geaNum, batNum],...
703                        engStaPreOptInx,geaStaPreOptInx, batStaPreOptInx);
704                end % end of ~inf(hamiltonian) if-statement
705            end %end of looping through all battery states
706            
707            fprintf('.');
708        end % end of looping through all gears
709    end % end of looping through all the current engine control states
710    fprintf('\n');

713    % approximation
714%     batEngPreInxVec = batStaLimBot : batStaLimTop;
715
716%     % find bounds for approximating next time index's batEng bounds
717%     batStaIdxBounds = find(~isinf(min(min(cos2goActTn3))));
718%     if length(batStaIdxBounds) < 1
719%        fprintf('WARNING: batStaIdxBounds gone at time %i\n', timInx); 
720%     end
721%     batStaLimBot = min(batStaIdxBounds);
722%     batStaLimTop = max(batStaIdxBounds);
723%                 
724    % Speichern der Kosten für den nächsten Schleifendurchlauf
725    %   save cost as previous path_idx value for the next loop
726    cos2goPreTn3 = cos2goActTn3; 
727    
728    % Speichern der Krafstoffenergie fï¿½r den nï¿½chsten Schleifendurchlauf
729    %   save fuel energy value as previous path_idx value for the next loop

731
732    % optimale Kraftstoffenergie zum aktuellen Punkt
733    %   optimal fuel energy at current point - save current mat in tensor
734    fulEngOptTn4(:,:,:,timInx) = fulEngActTn3;

738    %   flux quantity applied over the interval
739    batPwrOptTn4(:,:,:,timInx-1) = batPwrOptTn3;

741

743    % emoTrq
744    emoTrqOptTn4(:,:,:,timInx) = emoTrqActTn3;

748    brkTrqOptTn4(:,:,:,timInx) = brkTrqActTn3;
