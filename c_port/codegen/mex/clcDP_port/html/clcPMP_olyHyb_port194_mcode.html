 78%% Initialisieren der allgemein benÃ¶tigten KenngrÃ¶ÃŸen

 81% mittlere kinetische Energie im Wegschritt berechnen

 87
 88%% vorzeitiger Funktionsabbruch?
 89%   premature function termination?
 90% Drehzahl der Kurbelwelle und Grenzen
 91%   crankshaft speed and limits
 92
 93% Aus den kinetischen Energien des Fahrzeugs wird Ã¼ber die Raddrehzahl
 94% und die Ã¼bersetzung vom Getriebe die Kurbelwellendrehzahl berechnet.
 95% Zeilenrichtung entspricht den GÃ¤ngen. (Zeilenvektor)

100    sqrt(2*[engKinPre,engKinAct]/fzg_scalar.vehMas) / (fzg_scalar.whlDrr);
101
102% Abbruch, wenn die Drehzahlen der Kurbelwelle zu hoch im hybridischen

107end
108
109% Falls die Drehzahl des Verbrennungsmotors niedriger als die
110% Leerlaufdrehzahl ist,
111%   stop if crankshaft rotional speed is lower than the idling speed
112if iceFlg && any(crsSpdVec < crsSpdHybMin)
113    return;
114end
115
116% PrÃ¼fen, ob die Drehzahlgrenze des Elektromotors eingehalten wird
117%   check if electric motor speed limit is maintained
118if ~iceFlg && any(crsSpdVec > crsSpdEmoMax)
119    return;
120end

122% mittlere Kurbelwellendrehzahlen berechnen
123%   calculate average crankshaft rotational speed
124%   - really just selecting the previous path_idx KE crankshaft speed
125crsSpd = crsSpdVec(1);

127%% LÃ¤ngsdynamik berechnen
128%   calculate longitundinal dynamics
129% Es wird eine konstante Beschleunigung angenommen, die im Wegschritt
130% wayStp das Fahrzeug von velPre auf velAct beschleunigt.
131%   constant acceleration assumed when transitioning from velPre to velAct
132%   for the selected wayStp path_idx step distance
133
134% Berechnen der konstanten Beschleunigung
135%   calculate the constant acceleration
136vehAcc = (engKinAct - engKinPre) / (fzg_scalar.vehMas*wayStp);
137
138% Aus der mittleren kinetischen Energie im Intervall, der mittleren
139% Steigung und dem Gang lÃ¤sst sich Ã¼ber die Fahrwiderstandsgleichung
140% die nÃ¶tige Fahrwiderstandskraft berechnen, die aufgebracht werden
141% muss, um diese zu realisieren.
142%   from the (avg) kinetic energy in the interval, the (avg) slope and
143%   transition can calculate the necessary traction force on the driving
144%   resistance equation (PART OF EQUATION 5)
145
146% Steigungskraft aus der mittleren Steigung berechnen (Skalar)
147%   gradiant force from the calculated (average) gradient
148vehFrcSlp = fzg_scalar.vehMas * 9.81 * sin(slp);
149
150% Rollreibungskraft berechnen (Skalar)
151%   calculated rolling friction force - not included in EQ 5???
152vehFrcRol = fzg_scalar.whlRosResCof*fzg_scalar.vehMas * 9.81 * cos(slp);
153% Luftwiderstandskraft berechnen (2*c_a/m * E_kin) (Skalar) 
154%   calculated air resistance force 
155vehFrcDrg = 2*fzg_scalar.drgCof/fzg_scalar.vehMas*engKin;
156
157%% Berechnung der minimalen kosten der Hamiltonfunktion
158%   Calculating the minimum cost of the Hamiltonian
159
160%% Berechnen der Kraft am Rad fÃ¼r Antriebsstrangmodus
161%   calculate the force on the wheel for the drivetrain mode
162
163% % dynamische Fahrzeugmasse bei Fahrzeugmotor an berechnen. Das
164% % heiÃŸt es werden TrÃ¤gheitsmoment von Verbrennungsmotor,
165% % Elektromotor und RÃ¤dern mit einbezogen.
166%   calculate dynamic vehicle mass with the vehicle engine (with the moment
167%   of intertia of the ICE, electric motor, and wheels)
168% vehMasDyn = (par.iceMoi_geaRat(gea) +...
169%     par.emoGeaMoi_geaRat(gea) + par.whlMoi)/par.whlDrr^2 ...
170%     + par.vehMas;
171
172% Radkraft berechnen (Beschleunigungskraft + Steigungskraft +
173% Rollwiderstandskraft + Luftwiderstandskraft)
174%   caluclating wheel forces (accerlation force + gradient force + rolling
175%   resistance + air resistance)    EQUATION 5
176whlFrc  = vehAcc*fzg_scalar.vehMas + vehFrcSlp + vehFrcRol + vehFrcDrg;
177%% GetriebeÃ¼bersetzung und -verlust
178%   gear ratio and loss
179
180% Das Drehmoment des Rades ergibt sich Ã¼ber den Radhalbmesser aus
181% der Fahrwiderstandskraft.
182%   the weel torque is obtained from the wheel radius of the rolling
183%   resistance force (torque = force * distance (in this case, radius)
184whlTrq = whlFrc*fzg_scalar.whlDrr;
185
186% Berechnung des Kurbelwellenmoments
187% Hier muss unterschieden werden, ob das Radmoment positiv oder
188% negativ ist, da nur ein einfacher Wirkungsgrad fÃ¼r das Getriebe
189% genutzt wird
190%   it's important to determine sign of crankshaft torque (positive or

192%   PART OF EQ4  <- perhaps reversed? not too sure
193if whlTrq < 0
194    crsTrq = whlTrq / fzg_array.geaRat(gea) * fzg_scalar.geaEfy;
195else
196    crsTrq = whlTrq / fzg_array.geaRat(gea) / fzg_scalar.geaEfy;
197end
198
199%% Verbrennungsmotor
200%   internal combustion engine
201
202% maximales Moment des Verbrennungsmotors berechnen
203%   calculate max torque of the engine (quadratic based on rotation speed)

206    + fzg_array.iceTrqMaxCof(3);
207
208% minimales Moment des Verbrennungsmotors berechnen
209%   calculating mimimum ICE moment
210iceTrqMin = fzg_array.iceTrqMinCof(1) * crsSpd^2 ...
211    + fzg_array.iceTrqMinCof(2) * crsSpd ...
212    + fzg_array.iceTrqMinCof(3);
213
214%% Elektromotor
215%   electric motor
216
217% maximales Moment, dass die E-Maschine liefern kann
218%   max torque that the electric motor can provide - from interpolation
219% emoTrqMaxPos = ...
220%     lininterp1(par.emoSpdMgd(1,:)',par.emoTrqMax_emoSpd,crsSpd);
221emoTrqMaxPos = ...
222    interp1q(fzg_array.emoSpdMgd(1,:)',fzg_array.emoTrqMax_emoSpd,crsSpd);
223
224% Die gÃ¼ltigen Kurbelwellenmomente mÃ¼ssen kleiner sein als das
225% Gesamtmoment von E-Motor und Verbrennungsmotor
226%   The valid crankshaft moments must be less than the total moment of the
227%   electric motor and the ICE.Otherwise, leave the function
228if crsTrq > iceTrqMax + emoTrqMaxPos; 
229    return;
230end

232%% %% Optimaler Momentensplit - Minimierung der Hamiltonfunktion
233%       optimum torque split - minimizing the Hamiltonian
234% Die Vorgehensweise ist Ã¤hnlich wie bei der ECMS. Es wird ein Vektor der
235% mÃ¶glichen BatterieenergieÃ¤nderungen aufgestellt. Aus diesen lÃ¤sst sich 
236% eine Batterieklemmleistung berechnen. Aus der Ã¼ber das
237% Kurbelwellenmoment, ein Elektromotormoment berechnet werden kann.
238% Ãœber das geforderte Kurbelwellenmoment, kann fÃ¼r jedes Moment des 
239% Elektromotors ein Moment des Verbrennungsmotors gefunden werden. FÃ¼r 
240% jedes Momentenpaar kann die Hamiltonfunktion berechnet werden.
241% Ausgegeben wird der minimale Wert der Hamiltonfunktion und die
242% durch das dabei verwendete Elektromotormoment verursachte
243% Batterieladungsänderung.
244%   The procedure is similar to ECMS. It's based on a vector of possible
245%   battery energy changes, from which a battery terminal power can be
246%   calculated.
247%   From the crankshaft torque, an electric motor torque can be
248%   calculated, and an engine torque can be found for every electric motor
249%   moment. 
250%   For every moment-pair the Hamiltonian can be calculated. It
251%   outputs the minimum Hamilotnian value and the battery charge change
252%   caused by the electric motor torque used.
253
254%% Elektromotor - Aufstellen des Batterienergievektors
255%   electric motor - positioning the battery energy vectors
256
257if batEngStp > 0
258    [ ...
259        batEngDltMin,...Skalar - Ã¤nderung der minimalen BatterieenergieÃ¤nderung
260        batEngDltMax... Skalar - Ã¤nderung der maximalen BatterieenergieÃ¤nderung
261        ] = ...

263        (...
264        wayStp,...      Skalar - Wegschrittweite
265        vehVel,...      Skalar - mittlere Geschwindigkeit im Intervall
266        batPwrAux,...   Skalar - Nebenverbraucherlast
267        batEng,...      Skalar - Batterieenergie
268        fzg_scalar,...  struct - Fahrzeugparameter - nur skalar
269        fzg_array,...   struct - Fahrzeugparameter - nur array
270        crsSpd,...      Skalar - crankshaft rotational speed
271        crsTrq,...      Skalar - crankshaft torque
272        iceTrqMin,...   Skalar - min ICE torque allowed
273        iceTrqMax,...   Skalar - max ICE torque allowed
274        emoTrqMaxPos... Skalar - max EM torque possible
275        );
276    
277    % Es werden 2 FÃ¤lle unterschieden:
278    %   there are 2 different cases
279    if batEngDltMin > 0 && batEngDltMax > 0 % battery will discharge
280        
281        %% konventionelles Bremsen + Rekuperieren
282        %   conventional brakes + recuperation
283        %
284        % set change in energy to discretized integer values w/ ceil and
285        % floor. This also helps for easy looping
286
287        % Konventionelles Bremsen wird ebenfalls untersucht.
288        % Hier liegt eventuell noch Beschleunigungspotential, da diese
289        % Zustandswechsel u.U. ausgeschlossen werden kÃ¶nnen.
290        % NOTE: u.U. = unter ÃœmstÃ¤nden = circumstances permitting
291        %   convetional breaks also investigated. An accelerating potential
292        %   is still possible, as these states can be excluded

296        %   increasing, then set the delta min to zero

298        batEngDltMaxInx = floor(batEngDltMax/batEngStp);
299
300    else        
301        batEngDltMinInx = ceil(batEngDltMin/batEngStp);
302        batEngDltMaxInx = floor(batEngDltMax/batEngStp);
303    end
304else
305    batEngDltMinInx = 0;

307end
308
309% you got a larger min change than a max change-you're out of bounds. Leave
310% the function
311if batEngDltMaxInx < batEngDltMinInx
312    return;
313end
314
315%% Schleife Ã¼ber alle Elektromotormomente
316%   now loop through all the electric-motor torques
317% why isn't batEngDltInx discretized wrt batEngStp??
318for batEngDltInx = batEngDltMinInx:batEngDltMaxInx 
319    batEngDlt = batEngDltInx * batEngStp;
320    batEngAct = batEng + batEngDlt;
321    % open circuit voltage over each iteration
322    batOcv = batEngAct*fzg_array.batOcvCof_batEng(1,1)+fzg_array.batOcvCof_batEng(1,2);
323    
324    [ ...
325        batPwr,...          Skalar fÃ¼r die Batterieleistung in W
326        fulFrc ...          Skalar Krafstoffkraft in N
327        ] = ...
328        fulEngClc_port...            FUNCTION CALL
329        (wayStp,...         Skalar fÃ¼r die Wegschrittweite in m,

331        batPwrAux,...       Nebenverbraucherlast

333        batEngDlt, ...      Skalar - Batterieenergieänderung,
334        crsSpd,...          Skalar - crankshaft speed at given path_idx
335        crsTrq,...          Skalar - crankshaft torque at given path_idx
336        iceTrqMin,...       Skalar - min ICE torque allowed
337        iceTrqMax,...       Skalar - max ICE torque
338        fzg_scalar,...      struct - Fahrzeugparameter - nur skalar
339        fzg_array...        struct - Fahrzeugparameter - nur array        
340        );
341    
342    batFrc = batFrcClc_port(...      FUNCTION CALL
343        batPwr,...          Skalar - Batterieklemmleistung
344        vehVel,...          Skalar - mittlere Geschwindigkeit im Intervall
345        fzg_scalar.batRstDch,...   Skalar - Entladewiderstand

349    
350    %% Hamiltonfunktion bestimmen
351    %   determine the hamiltonian
352    % Eq (29b)
353    [cosHamMin,optPreInx] = min([fulFrc ...

356    
357    % Wenn der aktuelle Punkt besser ist, als der in cosHamMin
358    % gespeicherte Wert, werden die AusgabegrÃ¶ÃŸen neu beschrieben.

360    %   then the output values are rewritten (selected prev. value is = 2)
361    if optPreInx == 1
362        batFrcOut = batFrc;
363        fulFrcOut = fulFrc;
364    end
365    
366end
